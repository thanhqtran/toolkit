\newenvironment{chapabstract}
{
    \begin{center}%
        \vspace*{1.5cm}
        \Large \bfseries Abstract \vspace{-.5em}\vspace{0pt}
    \end{center}
    \list{}{%
        \setlength{\leftmargin}{5mm}% <---------- CHANGE HERE
        \setlength{\rightmargin}{\leftmargin}%
    }%
    \item\relax
}
{\par}

% Setup for subappendices environment
\AtBeginEnvironment{subappendices}{%
    % Add the 'Appendices' entry in the TOC as a chapter
    %\addtocontents{toc}{\protect\addvspace{10pt}\bfseries Appendices\par} 
    \counterwithin{figure}{section}  % Reset figure counter within section
    \counterwithin{table}{section}   % Reset table counter within section
}

\AtEndEnvironment{subappendices}{%
    % Reset the counters after subappendices
    \counterwithout{figure}{section}
    \counterwithin{figure}{chapter}  % Restore figure counter within chapter
    \counterwithout{table}{section}
    \counterwithin{table}{chapter}   % Restore table counter within chapter
}
